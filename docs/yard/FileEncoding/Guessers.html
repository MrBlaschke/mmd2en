def self.default_set
  [].tap {|set|
    set << Cheap::CORE_BOM
    set << Cheap::MORE_BOM
    set << Platform::FILE
    set << Platform::APPLE_XATTR
    set << Expensive::ASCII
    set << Expensive::UTF
    set << Expensive::LATIN
  }
end
