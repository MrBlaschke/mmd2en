def guess(file)
  path   = File.expand_path(file.is_a?(File) ? file.path : file)
  output = @sh.run_command(@tool, *@args, path.shellescape, :2>/dev/null')
  validate_result(instance_exec(output, &@process)) if @sh.ok?
end
