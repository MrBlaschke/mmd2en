def strain(string)
  max_len = @max_chars || string.length
  string.strip!
  string.gsub!(/[#{ALWAYS_STRIP.join('')}#{@also_strip}]/, '')
  string = string[0...max_len-@ellipsis.length] << @ellipsis if string.length > max_len
  string if string.length >= @min_chars
end
