def self.find_iana_charset(name)
  iana_mappings = {
    'macintosh'    => 'macRoman',
    'unknown-8bit' => nil         # yes, this is actually a IANA code (MIBEnum 2079)
  }
  ruby_encoding = iana_mappings.fetch(String(name).downcase, name)
  ruby_encoding and self.find(ruby_encoding)
end
