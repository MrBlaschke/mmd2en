def run_script(*lines, get_note_path_for: nil)
  if target = get_note_path_for
    lines << %Q{try}
    lines << %Q{get (name of notebook of #{target}) & "\n" & (local id of #{target})}
    lines << %Q{end try}
  end
  super(tell_command, *lines, %Q{end tell})
end
